\newcommand{\CONNECT}[2]
{
    \draw [-, thick] (#1) -- (#2);
}

\newcommand{\CONNECTVH}[2]
{
    \draw
      let
        \p1=(#1),
        \p2=(#2)
      in
        [-, thick] (\x1, \y1) -- (\x1, \y2) -- (\x2, \y2);
}

\newcommand{\CONNECTHV}[2]
{
    \draw
      let
        \p1=(#1),
        \p2=(#2)
      in
        [-, thick] (\x1, \y1) -- (\x2, \y1) -- (\x2, \y2);
}

\newcommand{\DOT}[2]
{
    \filldraw (#1) circle (0.3);
}
